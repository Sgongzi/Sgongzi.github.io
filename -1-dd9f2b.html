def func_grad(mu,sigma2):
    x = np.array([2,2.3,4,4,3,3,3.6,3,3.2,2.9])
    n = x.shape[0]
    dmu = np.sum(x-mu)/sigma2
    dsigma2 = -n/(2*sigma2)+np.sum((x-mu)**2)/(2*sigma2**2)
    grad = np.array([dmu,dsigma2])
    return grad

#计算Hession矩阵

def func_Hession(mu,sigma2):
    x = np.array([2,2.3,4,4,3,3,3.6,3,3.2,2.9])
    n = x.shape[0]
    

    d11 = -n/sigma2
    d12 = -np.sum(x-mu)/sigma2**2
    d21 = -np.sum(x-mu)/sigma2
    d22 = n/(2*sigma2**2)-np.sum((x-mu)**2)/sigma2**3
    return np.array([[d11,d12],[d21,d22]])

def newton_iter(mu,sigma2,func1,func2,max_iter):
    mu_init = mu
    sigma_init = sigma2
    x_init = np.array([mu_init,sigma_init])
    count = 0
    while (count<=max_iter) :
        hess = func2(mu_init,sigma_init)
        #求海塞矩阵的逆
        hess_inv = np.linalg.inv(hess)
        grad = func1(mu_init,sigma_init)
        x_next = x_init - np.dot(hess_inv,grad)
        x_init = x_next
        mu_init = x_next[0]
        sigma_init = x_next[1]
        count += 1
    return x_init
    
newton_iter(2.5,0.5,func_grad,func_Hession,30)
